# Command to install fastapi and it's peer dependencies:
==> pip install fastapi[all]


# Command to install uvicorn:
==> pip install uvicorn


# command to run the uvicorn web server
==> eg: uvicorn app.main:app --reload
OR
==> uvicorn app.main:app --host localhost --port 8000 --reload


# Command to install SQLAlchemy (ORM library):
==> pip install SQLAlchemy


# Command for installing database driver for postgresql ==> psycopg2  or psycopg2-binary
==> sudo apt-get install libpq-dev
==> sudo apt-get install build-essential

==> pip install psycopg2
OR
==> pip install psycopg2-binary


# Follow the steps below to install the UUID extension in the posgtres database:

Step 1: Access the running Postgres database with this command
# syntax ==> psql -U admin <database name>
# eg ==> psql -U admin fastapi_auth

Step 2: Execute this command to install the uuid-ossp plugin
==> CREATE EXTENSION IF NOT EXISTS "uuid-ossp";


\q to exit


# Command to install python package python-dotenv:
==> pip install python-dotenv


# Command to install python package alembic for creating migrations file:
==> pip install alembic


# Command to install passlib for hashing/encrypting password using bcrypt:
==> pip install "passlib[bcrypt]"
